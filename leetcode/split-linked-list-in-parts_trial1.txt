# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def splitListToParts(self, head: Optional[ListNode], k: int) -> List[Optional[ListNode]]:
        length, curr = 0, head

        while curr:
            length += 1
            curr = curr.next
        
        part_size = length // k
        larger_parts = length % k

        result = []
        curr = head

        for i in range(k):
            sublist_size = part_size + 1 if i < larger_parts else part_size
            if sublist_size == 0:
                result.append(None)
            else:
                sublist_head = curr
                for _ in range(sublist_size - 1):
                    curr = curr.next
                next_node = curr.next
                curr.next = None

                result.append(sublist_head)
                curr = next_node
        
        return result
